37template <>
38heffte::plan_options from_json<heffte::plan_options>(const json &j) {
39 heffte::plan_options options =
40 heffte::default_options<heffte::backend::fftw>();
41 if (j.contains("use_reorder")) {
42 options.use_reorder = j["use_reorder"];
43 }
44 if (j.contains("reshape_algorithm")) {
45 if (j["reshape_algorithm"] == "alltoall") {
46 options.algorithm = heffte::reshape_algorithm::alltoall;
47 } else if (j["reshape_algorithm"] == "alltoallv") {
48 options.algorithm = heffte::reshape_algorithm::alltoallv;
49 } else if (j["reshape_algorithm"] == "p2p") {
50 options.algorithm = heffte::reshape_algorithm::p2p;
51 } else if (j["reshape_algorithm"] == "p2p_plined") {
52 options.algorithm = heffte::reshape_algorithm::p2p_plined;
53 } else {
54 std::cerr << "Unknown communcation model " << j["reshape_algorithm"]
55 << std::endl;
56 }
57 }
58 if (j.contains("use_pencils")) {
59 options.use_pencils = j["use_pencils"];
60 }
61 if (j.contains("use_gpu_aware")) {
62 options.use_gpu_aware = j["use_gpu_aware"];
63 }
64 std::cout << "backend options: " << options << "\n\n";
65 return options;
66}

86template <>
87World from_json<World>(const json &j) {
88 int Lx = 0, Ly = 0, Lz = 0;
89 double dx = 0.0, dy = 0.0, dz = 0.0;
90 double x0 = 0.0, y0 = 0.0, z0 = 0.0;
91 std::string origo;
92
93 if (!j.count("Lx") || !j["Lx"].is_number_integer()) {
94 throw std::invalid_argument("Missing or invalid 'Lx' field in JSON input.");
95 }
96 Lx = j["Lx"];
97
98 if (!j.count("Ly") || !j["Ly"].is_number_integer()) {
99 throw std::invalid_argument("Missing or invalid 'Ly' field in JSON input.");
100 }
101 Ly = j["Ly"];
102
103 if (!j.count("Lz") || !j["Lz"].is_number_integer()) {
104 throw std::invalid_argument("Missing or invalid 'Lz' field in JSON input.");
105 }
106 Lz = j["Lz"];
107
108 if (!j.count("dx") || !j["dx"].is_number_float()) {
109 throw std::invalid_argument("Missing or invalid 'dx' field in JSON input.");
110 }
111 dx = j["dx"];
112
113 if (!j.count("dy") || !j["dy"].is_number_float()) {
114 throw std::invalid_argument("Missing or invalid 'dy' field in JSON input.");
115 }
116 dy = j["dy"];
117
118 if (!j.count("dz") || !j["dz"].is_number_float()) {
119 throw std::invalid_argument("Missing or invalid 'dz' field in JSON input.");
120 }
121 dz = j["dz"];
122
123 if (!j.count("origo") || !j["origo"].is_string()) {
124 throw std::invalid_argument(
125 "Missing or invalid 'origo' field in JSON input.");
126 }
127 origo = j["origo"];
128
129 if (origo != "center" && origo != "corner") {
130 throw std::invalid_argument("Invalid 'origo' field in JSON input.");
131 }
132
133 if (origo == "center") {
134 x0 = -0.5 * dx * Lx;
135 y0 = -0.5 * dy * Ly;
136 z0 = -0.5 * dz * Lz;
137 }
138
139 World world({Lx, Ly, Lz}, {x0, y0, z0}, {dx, dy, dz});
140
141 return world;
142}

612 int main() {
613 std::cout << "Reading configuration from json file:" << std::endl;
614 std::cout << m_settings.dump(4) << "\n\n";
615
616 World world(ui::from_json<World>(m_settings));
617 std::cout << "World: " << world << std::endl;
618
619 Decomposition decomp(world, m_comm);
620 auto plan_options =
621 ui::from_json<heffte::plan_options>(m_settings["plan_options"]);
622 FFT fft(decomp, m_comm, plan_options);
623 Time time(ui::from_json<Time>(m_settings));
624 ConcreteModel model;
625 model.set_fft(fft);
626 Simulator simulator(model, time);
627
628 if (m_settings.contains("model") &&
629 m_settings["model"].contains("params")) {
630 from_json(m_settings["model"]["params"], model);
631 }
632 read_detailed_timing_configuration();
633
634 std::cout << "Initializing model... " << std::endl;
635 model.initialize(time.get_dt());
636
637 add_result_writers(simulator);
638 add_initial_conditions(simulator);
639 add_boundary_conditions(simulator);
640
641 if (m_settings.contains("simulator")) {
642 const json &j = m_settings["simulator"];
643 if (j.contains("result_counter")) {
644 if (!j["result_counter"].is_number_integer()) {
645 throw std::invalid_argument(
646 "Invalid JSON input: missing or invalid 'result_counter' field.");
647 }
648 int result_counter = (int)j["result_counter"] + 1;
649 simulator.set_result_counter(result_counter);
650 }
651 if (j.contains("increment")) {
652 if (!j["increment"].is_number_integer()) {
653 throw std::invalid_argument(
654 "Invalid JSON input: missing or invalid 'increment' field.");
655 }
656 int increment = j["increment"];
657 time.set_increment(increment);
658 }
659 }
660
661 std::cout << "Applying initial conditions" << std::endl;
662 simulator.apply_initial_conditions();
663 if (time.get_increment() == 0) {
664 std::cout << "First increment: apply boundary conditions" << std::endl;
665 simulator.apply_boundary_conditions();
666 simulator.write_results();
667 }
668
669 while (!time.done()) {
670 time.next(); // increase increment counter by 1
671 simulator.apply_boundary_conditions();
672
673 double l_steptime = 0.0; // l = local for this mpi process
674 double l_fft_time = 0.0;
675 MPI_Barrier(m_comm);
676 l_steptime = -MPI_Wtime();
677 model.step(time.get_current());
678 MPI_Barrier(m_comm);
679 l_steptime += MPI_Wtime();
680 l_fft_time = fft.get_fft_time();
681
682 if (m_detailed_timing) {
683 double timing[2] = {l_steptime, l_fft_time};
684 MPI_Send(timing, 2, MPI_DOUBLE, 0, 42, m_comm);
685 if (m_worker.get_rank() == 0) {
686 int num_ranks = m_worker.get_num_ranks();
687 double timing[num_ranks][2];
688 for (int rank = 0; rank < num_ranks; rank++) {
689 MPI_Recv(timing[rank], 2, MPI_DOUBLE, rank, 42, m_comm,
690 MPI_STATUS_IGNORE);
691 }
692 auto inc = time.get_increment();
693 if (m_detailed_timing_print) {
694 auto old_precision = std::cout.precision(6);
695 std::cout << "Timing information for all processes:" << std::endl;
696 std::cout << "step;rank;step_time;fft_time" << std::endl;
697 for (int rank = 0; rank < num_ranks; rank++) {
698 std::cout << inc << ";" << rank << ";" << timing[rank][0] << ";"
699 << timing[rank][1] << std::endl;
700 }
701 std::cout.precision(old_precision);
702 }
703 if (m_detailed_timing_write) {
704 // so we end up to a binary file, and opening with e.g. Python
705 // np.fromfile("timing.bin").reshape(n_steps, n_procs, 2)
706 std::ofstream outfile(m_detailed_timing_filename, std::ios::app);
707 outfile.write((const char *)timing, sizeof(double) * 2 * num_ranks);
708 outfile.close();
709 }
710 }
711 }
712
713 // max reduction over all mpi processes
714 MPI_Reduce(&l_steptime, &m_steptime, 1, MPI_DOUBLE, MPI_MAX, 0, m_comm);
715 MPI_Reduce(&l_fft_time, &m_fft_time, 1, MPI_DOUBLE, MPI_MAX, 0, m_comm);
716
717 if (time.do_save()) {
718 simulator.apply_boundary_conditions();
719 simulator.write_results();
720 }
721
722 // Calculate eta from average step time.
723 // Use exponential moving average when steps > 3.
724 m_avg_steptime = m_steptime;
725 if (m_steps_done > 3) {
726 m_avg_steptime = 0.01 * m_steptime + 0.99 * m_avg_steptime;
727 }
728 int increment = time.get_increment();
729 double t = time.get_current(), t1 = time.get_t1();
730 double eta_i = (t1 - t) / time.get_dt();
731 double eta_t = eta_i * m_avg_steptime;
732 double other_time = m_steptime - m_fft_time;
733 std::cout << "Step " << increment << " done in " << m_steptime << " s ";
734 std::cout << "(" << m_fft_time << " s FFT, " << other_time
735 << " s other). ";
736 std::cout << "Simulation time: " << t << " / " << t1;
737 std::cout << " (" << (t / t1 * 100) << " % done). ";
738 std::cout << "ETA: " << pfc::utils::TimeLeft(eta_t) << std::endl;
739
740 m_total_steptime += m_steptime;
741 m_total_fft_time += m_fft_time;
742 m_steps_done += 1;
743 }
744
745 double avg_steptime = m_total_steptime / m_steps_done;
746 double avg_fft_time = m_total_fft_time / m_steps_done;
747 double avg_oth_time = avg_steptime - avg_fft_time;
748 double p_fft = avg_fft_time / avg_steptime * 100.0;
