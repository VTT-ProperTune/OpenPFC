161 const int Lz = s.Lz;

163 MPI_Comm comm = MPI_COMM_WORLD;

181 MPI_Datatype filetype;
182 const int size_array[] = {Lx, Ly, Lz};
183 const int subsize_array[] = {inbox_high[0] - inbox_low[0] + 1,
184 inbox_high[1] - inbox_low[1] + 1,
185 inbox_high[2] - inbox_low[2] + 1};
186 const int start_array[] = {inbox_low[0], inbox_low[1], inbox_low[2]};
187 MPI_Type_create_subarray(3, size_array, subsize_array, start_array,
188 MPI_ORDER_FORTRAN, MPI_DOUBLE, &filetype);

197 std::cout << std::endl
198 << "***** MEMORY ALLOCATION STATUS *****" << std::endl;

200 s.allocate(s.fft.size_inbox(), s.fft.size_outbox());
201 const size_t mem_allocated = s.mem_allocated;
202 const size_t mem_allocated_wrk =
203 sizeof(std::complex<double>) * s.fft.size_workspace();
204 MPI_Send(&mem_allocated, 1, MPI_LONG_INT, 0, 0, MPI_COMM_WORLD);
205 MPI_Send(&mem_allocated_wrk, 1, MPI_LONG_INT, 0, 1, MPI_COMM_WORLD);
206 MPI_Barrier(MPI_COMM_WORLD);
207 if (me == 0) {
208 size_t size = 0;
209 size_t size_wrk = 0;
210 size_t total_size = 0;
211 size_t total_size_wrk = 0;
212 for (int i = 0; i < num_ranks; i++) {
213 MPI_Recv(&size, 1, MPI_LONG_INT, i, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
214 MPI_Recv(&size_wrk, 1, MPI_LONG_INT, i, 1, MPI_COMM_WORLD,
215 MPI_STATUS_IGNORE);
216 std::cout << "MPI Worker " << i << ", " << (size + size_wrk)
217 << " bytes allocated (" << size << " bytes user data, "
218 << size_wrk << " bytes for fft workspace)" << std::endl;
219 total_size += size;
220 total_size_wrk += size_wrk;
221 }
222 const long int ndofs = s.Lx * s.Ly * s.Lz;
223 const double d = 1.0 / ndofs;
224 const double size_perdof = d * total_size;
225 const double size_wrk_perdof = d * total_size_wrk;
226 const double size_total_perdof = d * (total_size + total_size_wrk);
227 std::cout << "Total " << total_size << " bytes allocated for user data ("
228 << size_perdof << " bytes / dof)" << std::endl;
229 std::cout << "Total " << total_size_wrk
230 << " bytes allocated for workspace (" << size_wrk_perdof
231 << " bytes / dof)" << std::endl;
232 std::cout << "Total " << (total_size + total_size_wrk)
233 << " bytes allocated (" << size_total_perdof << " bytes / dof)"
234 << std::endl;
235 std::cout << std::endl;
236 }

276
277 double Sw = 0.0;

280 MPI_Barrier(MPI_COMM_WORLD);
281 s.write_results(n, t);
282 MPI_Barrier(MPI_COMM_WORLD);
283 Sw += MPI_Wtime();
284 if (me == 0) {

287 }

289 if (me == 0) {
290 std::cout << std::endl

292 << std::endl;

295 auto start = std::chrono::high_resolution_clock::now();
296
297 // for timing step time
298 const double alpha = 0.05;
299 double S = 0.0;
300 std::array<double, 8> timing;
301 int t_saveat_cnt = 1;
302 double t_saveat = s.t0 + t_saveat_cnt * s.get_saveat();

304 while (!s.done(n, t)) {
305 n += 1;
306 t += s.get_dt(n, t);

308 if (t >= t_saveat) {

312 if (me == 0) {

315 }

317 double dt_step = -MPI_Wtime();
318 MPI_Barrier(MPI_COMM_WORLD);
319 s.apply_bc(inbox_low, inbox_high);
320 MPI_Barrier(MPI_COMM_WORLD);
321 s.step(n, t);
322 MPI_Barrier(MPI_COMM_WORLD);
323 MPI_Send(&(s.timing), 8, MPI_DOUBLE, 0, MPI_TAG_TIMING, MPI_COMM_WORLD);
324 MPI_Barrier(MPI_COMM_WORLD);
325 dt_step += MPI_Wtime();
326
327 if (me == 0) {
328
329 std::cout << "Step finished. Timing information:" << std::endl;
330 double total_time = 0.0;
331 double fft_time = 0.0;
332 double other_time = 0.0;
333 for (int i = 0; i < num_ranks; i++) {
334 MPI_Recv(&timing, 8, MPI_DOUBLE, i, MPI_TAG_TIMING, MPI_COMM_WORLD,
335 MPI_STATUS_IGNORE);
336#ifdef PFC_SHOW_TIMING_PER_WORKER
337 std::cout << "MPI Worker " << i << ": FFT " << timing[1] << ", Other "
338 << timing[2] << ", Total " << timing[0] << std::endl;
339#endif
340 total_time += timing[0];
341 fft_time += timing[1];
342 other_time += timing[2];
343 }
344 total_time /= num_ranks;
345 fft_time /= num_ranks;
346 other_time /= num_ranks;
347 std::cout << "Average time: FFT " << fft_time << ", Other " << other_time
348 << ", Total " << total_time << std::endl;
349
350 S = (n < 5) ? dt_step : alpha * dt_step + (1.0 - alpha) * S;
351 auto n_left = (s.t1 - t) / s.get_dt(n, t);
352 auto eta = S * n_left;
353 std::cout << "Step execution time: " << dt_step
354 << " seconds. Average step execution time: " << S << " seconds."
355 << std::endl;
356 auto pct_done = 100.0 * n / (n + n_left);
357 std::cout << "Simulation time: " << t << " seconds. " << n
358 << " of estimated " << (n + n_left) << " steps (" << pct_done
359 << " %) done." << std::endl;
360 std::cout << "Simulation is estimated to be ready in " << eta
361 << " seconds." << std::endl;
362 }
363
364 if (write_results) {
365 double dt_write = -MPI_Wtime();
366 MPI_Barrier(MPI_COMM_WORLD);
367 s.write_results(t_saveat_cnt, t);
368 MPI_Barrier(MPI_COMM_WORLD);
369 dt_write += MPI_Wtime();
370 if (me == 0) {
371 Sw = alpha * dt_write + (1.0 - alpha) * Sw;
372 std::cout << "Results writing time: " << dt_write
373 << " seconds (avg: " << Sw << " seconds)" << std::endl;
374 }
375 t_saveat_cnt += 1;
376 t_saveat = s.t0 + t_saveat_cnt * s.get_saveat();
377 }
378
379 if (me == 0) {
380 std::cout << "***** FINISHING STEP # " << n << " *****" << std::endl
381 << std::endl;
382 std::cout.flush();
383 }
384 }
385 auto stop = std::chrono::high_resolution_clock::now();
386 auto duration =
387 std::chrono::duration_cast<std::chrono::milliseconds>(stop - start);
388 if (me == 0) {
389 std::cout << n << " iterations in " << duration.count() / 1000.0
390 << " seconds (" << duration.count() / n << " ms / iteration)"
391 << std::endl;
392 }
393 if (me == 0) {
394 std::cout << "Simulation done. Status message: " + s.status_msg
395 << std::endl;
396 }
